def self.create_from_omniauth(auth)
  create! do |user|
    user.user_type_id = UserType.find_by_name(auth['provider']).id
    user.uid = auth["uid"]
    user.email = auth['info']['email']
    user.gravatar_email = auth['info']['email']
    user.name = auth["info"]["name"]
  end
end
